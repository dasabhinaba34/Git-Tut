git --version // to check the version
git config --global user.name "abhinaba" // configuring username
git config --globaluser.email "you@example.com" // configuring email
git init // to initialize a git repo
git status // to check the status of the repo
git add <file-name> // to add the file in staging area
git commit -m "<message>" // to commit with a message
git log // to check all the commits
git checkout <commit address>/<branch-name> //to go to the particular commit/branch
git checkout master // to go to the master  
touch .gitignore // to create a .gitignore file
git merge <branch-name> // to merge a branch with the current branch
git remote add origin <link> // to add remote repository link
git branch -M <branch-name> // to create a new branch 
git push -u origin master // to push the master branch
git push -u origin <branch-name> // to push a branch into GitHub
touch REMADME.md // to add a readme file
git clone <link> // to clone a repo
git remote set-url origin <repository http link> // to change or rename the remote repository
git remote -v  // to check the remote repository
git reset HEAD^ // to delete the very last commit
git push origin +HEAD # force-push the new HEAD commit
